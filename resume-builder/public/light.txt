            void cycler.offsetWidth; 
            cycler.classList.remove('prepare');

        let current = 0, target = 98, step = target / (2000 / 16); 
        setTimeout(() => {
            const updateCounter = () => {
                current += step;
                if (current < target) {
                    scoreEl.innerText = Math.floor(current);
                    requestAnimationFrame(updateCounter);
                } else { scoreEl.innerText = target; }
            };
            updateCounter();
        }, 500); 
    });

    let activeTut = 'editor';
    const wait = ms => new Promise(r => setTimeout(r, ms));

    function switchTut(view) {
        if(activeTut === view) return;
        activeTut = view;
        
        // Update Tabs
        document.getElementById('tab-editor').classList.toggle('active', view === 'editor');
        document.getElementById('tab-canvas').classList.toggle('active', view === 'canvas');
        
        // Update Windows
        document.getElementById('view-editor').classList.toggle('active', view === 'editor');
        document.getElementById('view-canvas').classList.toggle('active', view === 'canvas');
    }

    // Editor Loop
    async function loopEditor() {
        const cur = document.getElementById('ed-cursor'), input = document.getElementById('ed-input'), typed = document.getElementById('ed-typed'), prev = document.getElementById('ed-preview'), btn = document.getElementById('ed-btn'), bText = document.getElementById('ed-btn-text');
        
        while(true) {
            if(activeTut !== 'editor') { await wait(500); continue; }

            typed.innerText = ""; prev.innerText = "";
            btn.classList.remove('clicked'); bText.innerText = "Download PDF";
            cur.style.opacity = 1; input.classList.remove('active');
            cur.style.transform = `translate(250px, 150px)`; 
            await wait(1000);
            if(activeTut !== 'editor') continue;
            
            // Move to input
            cur.style.transform = `translate(90px, 120px)`; 
            await wait(800);
            if(activeTut !== 'editor') continue;

            cur.style.opacity = 0; await wait(300);
            input.classList.add('active'); // show caret
            
            // Type
            const text = "FullStackDeveloper";
            for(let i=0; i<text.length; i++) {
                if(activeTut !== 'editor') break;
                typed.innerText += text[i]; prev.innerText += text[i];
                await wait(70 + Math.random() * 50); 
            }
            if(activeTut !== 'editor') { input.classList.remove('active'); continue; }
            
            await wait(500);
            input.classList.remove('active'); cur.style.opacity = 1;
            
            // Move to btn
            cur.style.transform = `translate(90px, 190px)`; 
            await wait(800);
            if(activeTut !== 'editor') continue;
            
            // Click
            btn.classList.add('clicked'); bText.innerText = "Saved!";
            await wait(2500);
            cur.style.opacity = 0; await wait(1000);
        }
    }

    // Canvas Loop
    async function loopCanvas() {
        const cur = document.getElementById('cv-cursor'), tool = document.getElementById('cv-tool'), shape = document.getElementById('cv-shape'), name = document.getElementById('cv-name'), role = document.getElementById('cv-role');
        
        while(true) {
            if(activeTut !== 'canvas') { await wait(500); continue; }

            shape.className = "tc-el tc-shape"; name.className = "tc-el tc-name"; role.className = "tc-el tc-role";
            tool.classList.remove('active'); cur.style.opacity = 0;
            cur.style.transform = `translate(150px, 150px)`;
            await wait(1000);
            if(activeTut !== 'canvas') continue;

            // Select Tool
            cur.style.opacity = 1;
            cur.style.transform = `translate(20px, 30px)`;
            await wait(800);
            cur.style.transform += ' scale(0.85)'; await wait(200);
            cur.style.transform = cur.style.transform.replace(' scale(0.85)', '');
            tool.classList.add('active'); shape.classList.add('show');
            await wait(500); tool.classList.remove('active');
            if(activeTut !== 'canvas') continue;

            // Drag to top
            cur.style.transform = `translate(150px, 140px)`;
            await wait(800);
            cur.style.transform += ' scale(0.85)'; shape.classList.add('selected'); await wait(200);
            shape.classList.add('dragged');
            cur.style.transform = `translate(90px, 30px) scale(0.85)`;
            await wait(800);
            cur.style.transform = cur.style.transform.replace(' scale(0.85)', ''); await wait(300);
            if(activeTut !== 'canvas') continue;

            // Resize
            cur.style.transform = `translate(140px, 70px)`;
            await wait(800);
            cur.style.transform += ' scale(0.85)'; shape.classList.add('resized');
            cur.style.transform = `translate(250px, 70px) scale(0.85)`;
            await wait(800);
            cur.style.transform = cur.style.transform.replace(' scale(0.85)', ''); shape.classList.remove('selected'); await wait(400);
            if(activeTut !== 'canvas') continue;

            // Move Text
            cur.style.transform = `translate(100px, 140px)`;
            await wait(800);
            cur.style.transform += ' scale(0.85)'; name.classList.add('moved'); role.classList.add('moved');
            cur.style.transform = `translate(100px, 50px) scale(0.85)`;
            await wait(800);
            cur.style.transform = cur.style.transform.replace(' scale(0.85)', '');
            
            await wait(2500);
            cur.style.opacity = 0; await wait(1000);
        }
    }

    if(window.innerWidth > 900) {
        // Start loops in background, they pause automatically based on activeTut
        setTimeout(() => {
            loopEditor();
            loopCanvas();
        }, 1000);
    }

        async function fetchSavedResumes() {
            const userId = localStorage.getItem('userId');
            const container = document.getElementById('saved-resumes-container');
            
            if (!userId) {
                if(container) container.innerHTML = '<p style="color: var(--text-dim); font-size: 0.85rem;">Please log in to view saved resumes.</p>';
                return;
            }

            try {
                const response = await fetch(`http://localhost:5000/api/resume/user/${userId}`);
                const result = await response.json();

                if(container) container.innerHTML = '';
                if (response.ok && result.data && result.data.length > 0 && container) {
                    result.data.forEach((resume, index) => {
                        const template = document.getElementById('saved-card-template');
                        const clone = template.content.cloneNode(true);
                        
                        // Add animation delay for cascading entrance
                        const cardElement = clone.querySelector('.saved-card');
                        cardElement.style.animationDelay = `${index * 0.1}s`;
                        
                        clone.querySelector('.resume-title').textContent = resume.personalInfo?.name || 'Untitled Resume';
                        
                        const date = new Date(resume.updatedAt || resume.createdAt).toLocaleDateString();
                        clone.querySelector('.resume-date').textContent = `Last updated: ${date}`;
                        
                        clone.querySelector('.edit-btn').onclick = () => {
                            if(resume.templateSpecificInputs && resume.templateSpecificInputs.isCustom) {
                                window.location.href = `customize.html?id=${resume._id}`;
                            } else {
                                window.location.href = `editor.html?id=${resume._id}`;
                            }
                        };
                        
                        clone.querySelector('.delete-btn').onclick = async (e) => {
                            e.stopPropagation();
                            if(confirm("Are you sure you want to delete this resume?")) {
                                try {
                                    await fetch(`http://localhost:5000/api/resume/${resume._id}`, { method: 'DELETE' });
                                    fetchSavedResumes(); // reload
                                } catch(err) {
                                    alert("Failed to delete.");
                                }
                            }
                        };
                        
                        container.appendChild(clone);
                    });
                } else if(container) {
                    container.innerHTML = '<p style="color: var(--text-dim); font-size: 0.85rem;">No saved resumes found.</p>';
                }
            } catch (error) {
                console.error("Error fetching resumes:", error);
                if(container) container.innerHTML = '<p style="color: #ef4444; font-size: 0.85rem;">Error loading saved resumes. Is your server running?</p>';
            }
        }

        window.addEventListener('DOMContentLoaded', () => {

        async function fetchSavedResumes() {
            const userId = localStorage.getItem('userId');
            const container = document.getElementById('saved-resumes-container');
            
            if (!userId) {
                if(container) container.innerHTML = '<p style="color: var(--text-dim); font-size: 0.85rem;">Please log in to view saved resumes.</p>';
                return;
            }

            try {
                const response = await fetch(`http://localhost:5000/api/resume/user/${userId}`);
                const result = await response.json();

                if(container) container.innerHTML = '';
                if (response.ok && result.data && result.data.length > 0 && container) {
                    result.data.forEach((resume, index) => {
                        const template = document.getElementById('saved-card-template');
                        const clone = template.content.cloneNode(true);
                        
                        // Add animation delay for cascading entrance
                        const cardElement = clone.querySelector('.saved-card');
                        cardElement.style.animationDelay = `${index * 0.1}s`;
                        
                        clone.querySelector('.resume-title').textContent = resume.personalInfo?.name || 'Untitled Resume';
                        
                        const date = new Date(resume.updatedAt || resume.createdAt).toLocaleDateString();
                        clone.querySelector('.resume-date').textContent = `Last updated: ${date}`;
                        
                        clone.querySelector('.edit-btn').onclick = () => {
                            if(resume.templateSpecificInputs && resume.templateSpecificInputs.isCustom) {
                                window.location.href = `customize.html?id=${resume._id}`;
                            } else {
                                window.location.href = `editor.html?id=${resume._id}`;
                            }
                        };
                        
                        clone.querySelector('.delete-btn').onclick = async (e) => {
                            e.stopPropagation();
                            if(confirm("Are you sure you want to delete this resume?")) {
                                try {
                                    await fetch(`http://localhost:5000/api/resume/${resume._id}`, { method: 'DELETE' });
                                    fetchSavedResumes(); // reload
                                } catch(err) {
                                    alert("Failed to delete.");
                                }
                            }
                        };
                        
                        container.appendChild(clone);
                    });
                } else if(container) {
                    container.innerHTML = '<p style="color: var(--text-dim); font-size: 0.85rem;">No saved resumes found.</p>';
                }
            } catch (error) {
                console.error("Error fetching resumes:", error);
                if(container) container.innerHTML = '<p style="color: #ef4444; font-size: 0.85rem;">Error loading saved resumes. Is your server running?</p>';
            }
        }
