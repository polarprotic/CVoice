    // ── HISTORY ──
    function saveState() {
        if (historyIndex < historyStack.length - 1) historyStack = historyStack.slice(0, historyIndex + 1);
        const clone = document.getElementById('page-1').cloneNode(true);
        clone.querySelectorAll('.selected, .editing, .multi-selected').forEach(el => {

        const ws = document.getElementById('workspace');

    function setZIndex(mode) {
        if (!selectedElement) return;
        const siblings = Array.from(selectedElement.parentElement.querySelectorAll('.element')).filter(e => e !== selectedElement);
        const zValues = siblings.map(e => parseInt(e.style.zIndex) || 1);
        if (mode === 'front') {
            const maxZ = zValues.length ? Math.max(...zValues) : highestZIndex;
            const newZ = maxZ + 1;
            selectedElement.style.zIndex = newZ;
            highestZIndex = newZ;
        } else if (mode === 'back') {
            const minZ = zValues.length ? Math.min(...zValues) : 1;
            selectedElement.style.zIndex = minZ - 1;
        }
        syncZDisplay();
        saveState();
    }

        const allSiblings = Array.from((multiSelected[0] || document.getElementById('page-1')).parentElement.querySelectorAll('.element')).filter(e => !multiSelected.includes(e));
        const zValues = allSiblings.map(e => parseInt(e.style.zIndex) || 1);
        if (mode === 'front') {
            const maxZ = zValues.length ? Math.max(...zValues) : highestZIndex;
            multiSelected.forEach((el, i) => { el.style.zIndex = maxZ + 1 + i; });

            multiSelected.forEach((el, i) => { el.style.zIndex = minZ - multiSelected.length + i; });

    // ── SUPER SYNC TOOLBAR ──
    function syncToolbar() {
        if (!selectedElement || !selectedElement.classList.contains('text-element')) return;
        const isEditing = selectedElement.classList.contains('editing');
        let targetNode = selectedElement;
        if (isEditing) {
            const sel = window.getSelection();
            if (sel.rangeCount > 0 && selectedElement.contains(sel.anchorNode)) {
                targetNode = sel.anchorNode.nodeType === 3 ? sel.anchorNode.parentNode : sel.anchorNode;
            }
        }
        const computed = window.getComputedStyle(targetNode);
        const align = computed.textAlign || 'left';
        ['left','center','right','justify'].forEach(a => {
            const btn = document.getElementById('align-' + a);
            if (btn) btn.classList.toggle('active', align === a);
        });
        const wsBtn = document.getElementById('align-wordspacing');
        if (wsBtn) {
            const ws = computed.wordSpacing;
            wsBtn.classList.toggle('active', ws && ws !== 'normal' && ws !== '0px' && ws !== '');
        }
        const boldBtn = document.getElementById('format-bold');
        const italicBtn = document.getElementById('format-italic');
        const underlineBtn = document.getElementById('format-underline');
        if (isEditing) {
            if (boldBtn) boldBtn.classList.toggle('active', document.queryCommandState('bold'));
            if (italicBtn) italicBtn.classList.toggle('active', document.queryCommandState('italic'));
            if (underlineBtn) underlineBtn.classList.toggle('active', document.queryCommandState('underline'));
        } else {
            const isBold = computed.fontWeight === 'bold' || parseInt(computed.fontWeight) >= 600;
            const isItalic = computed.fontStyle === 'italic';
            const isUnderline = computed.textDecorationLine && computed.textDecorationLine.includes('underline');
            if (boldBtn) boldBtn.classList.toggle('active', isBold);
            if (italicBtn) italicBtn.classList.toggle('active', isItalic);
            if (underlineBtn) underlineBtn.classList.toggle('active', isUnderline);
        }
        const ulBtn = document.getElementById('format-ul');
        const olBtn = document.getElementById('format-ol');
        if (ulBtn && olBtn) {
            let isUl = false, isOl = false;
            if (isEditing) { isUl = document.queryCommandState('insertUnorderedList'); isOl = document.queryCommandState('insertOrderedList'); }
            else { isUl = !!selectedElement.querySelector('ul'); isOl = !!selectedElement.querySelector('ol'); }
            ulBtn.classList.toggle('active', isUl);
            olBtn.classList.toggle('active', isOl);
        }
        if (document.activeElement !== document.getElementById('font-size'))
            document.getElementById('font-size').value = parseInt(computed.fontSize);
        if (document.activeElement !== document.getElementById('line-height'))
            document.getElementById('line-height').value = computed.lineHeight !== 'normal' ? (parseFloat(computed.lineHeight) / parseFloat(computed.fontSize)).toFixed(1) : 1.5;
        if (document.activeElement !== document.getElementById('letter-spacing'))
            document.getElementById('letter-spacing').value = computed.letterSpacing !== 'normal' ? parseFloat(computed.letterSpacing) : 0;
        document.getElementById('color-picker').value = rgbToHex(computed.color);

        }

    function applyTextFormat(command, val = null, save = true) {
        if (!selectedElement || !selectedElement.classList.contains('text-element')) return;
        const isEditing = selectedElement.classList.contains('editing');
        if (isEditing) {
            const activeEl = document.activeElement;
            const isInspectorInput = activeEl && (activeEl.tagName === 'INPUT' || activeEl.tagName === 'SELECT' || activeEl.tagName === 'TEXTAREA');
            selectedElement.focus();
            const sel = window.getSelection();
            if (currentRange) { sel.removeAllRanges(); sel.addRange(currentRange); }
            if (command === 'fontSizePx') {
                document.execCommand('fontSize', false, '7');
                selectedElement.querySelectorAll('font[size="7"]').forEach(f => { f.removeAttribute('size'); f.style.fontSize = val + 'px'; });
            } else if (command === 'fontFamily') {
                document.execCommand('fontName', false, val.split(',')[0].replace(/['"]/g, ''));
            } else if (command === 'foreColor') {
                document.execCommand('foreColor', false, val);
            } else if (command !== 'unlink') {
                document.execCommand(command, false, val);
            }
            if (sel.rangeCount > 0) currentRange = sel.getRangeAt(0).cloneRange();
            if (isInspectorInput && activeEl) activeEl.focus();
        } else {
            if (command === 'fontSizePx') updateStyle('fontSize', val + 'px', false);
            if (command === 'fontFamily') updateStyle('fontFamily', val, false);
            if (command === 'foreColor') updateStyle('color', val, false);
            if (command === 'bold') { const w = window.getComputedStyle(selectedElement).fontWeight; updateStyle('fontWeight', w >= 600 ? '400' : '700', false); }
            if (command === 'italic') { const i = window.getComputedStyle(selectedElement).fontStyle; updateStyle('fontStyle', i === 'italic' ? 'normal' : 'italic', false); }
            if (command === 'underline') { const u = window.getComputedStyle(selectedElement).textDecorationLine; updateStyle('textDecoration', u.includes('underline') ? 'none' : 'underline', false); }
            if (command === 'unlink') {
                selectedElement.contentEditable = "true"; selectedElement.focus();
                document.execCommand('unlink', false, null);
                selectedElement.contentEditable = "false";
                window.getSelection().removeAllRanges();
            }
        }
        syncToolbar();
        if (save) saveState();
    }

    function insertList(type) {
        if (!selectedElement || !selectedElement.classList.contains('text-element')) return;
        const isEditing = selectedElement.classList.contains('editing');
        if (isEditing) {
            selectedElement.focus();
            const sel = window.getSelection();
            if (currentRange) { sel.removeAllRanges(); sel.addRange(currentRange); }
            const cmd = type === 'ul' ? 'insertUnorderedList' : 'insertOrderedList';
            document.execCommand(cmd, false, null);
            selectedElement.querySelectorAll('ul, ol').forEach(list => { list.style.paddingLeft = '18px'; list.style.margin = '0'; });
            selectedElement.querySelectorAll('li').forEach(li => { li.style.display = 'list-item'; });
            if (sel.rangeCount > 0) currentRange = sel.getRangeAt(0).cloneRange();
        } else {
            const tag = type === 'ul' ? 'ul' : 'ol';
            const otherTag = type === 'ul' ? 'ol' : 'ul';
            const existingList = selectedElement.querySelector(tag);
            const wrongList = selectedElement.querySelector(otherTag);
            if (existingList) {
                const items = Array.from(existingList.querySelectorAll('li'));
                selectedElement.innerHTML = items.map(li => { const html = li.innerHTML.trim(); return (html === '<br>' || html === '') ? '' : html; }).join('<br>');
            } else if (wrongList) {
                const items = Array.from(wrongList.querySelectorAll('li')).map(li => li.outerHTML).join('');
                selectedElement.innerHTML = `<${tag} style="padding-left:18px;margin:0;">${items}</${tag}>`;
            } else {
                let content = selectedElement.innerHTML.replace(/<div[^>]*>/gi, '<br>').replace(/<\/div>/gi, '').replace(/<p[^>]*>/gi, '<br>').replace(/<\/p>/gi, '');
                const lines = content.split(/<br\s*\/?>/gi);
                const items = lines.map(line => { const trimmed = line.trim(); return `<li>${trimmed === '' ? '<br>' : trimmed}</li>`; }).join('');
                selectedElement.innerHTML = `<${tag} style="padding-left:18px;margin:0;">${items}</${tag}>`;
            }
            selectedElement.querySelectorAll('li').forEach(li => { li.style.display = 'list-item'; });
        }
        syncToolbar(); saveState();
    }

    // ── SMART SNAP GUIDES ──
    function calculateSnap(el, newLeft, newTop, newWidth, newHeight) {
        const threshold = 5;
        let snappedX = false, snappedY = false, guideX = -1, guideY = -1;
        const others = Array.from(el.parentElement.querySelectorAll('.element')).filter(e => e !== el && !e.classList.contains('handle'));
        for (let other of others) {
            const oL = parseFloat(other.style.left)||0, oT = parseFloat(other.style.top)||0;
            const oW = other.offsetWidth, oH = other.offsetHeight;
            const oR = oL+oW, oB = oT+oH, oCX = oL+oW/2, oCY = oT+oH/2;
            const cR = newLeft+newWidth, cB = newTop+newHeight, cCX = newLeft+newWidth/2, cCY = newTop+newHeight/2;
            if (!snappedX) {
                if (Math.abs(newLeft-oL)<threshold) { newLeft=oL; guideX=oL; snappedX=true; }
                else if (Math.abs(cR-oR)<threshold) { newLeft=oR-newWidth; guideX=oR; snappedX=true; }
                else if (Math.abs(cCX-oCX)<threshold) { newLeft=oCX-newWidth/2; guideX=oCX; snappedX=true; }
                else if (Math.abs(newLeft-oR)<threshold) { newLeft=oR; guideX=oR; snappedX=true; }
                else if (Math.abs(cR-oL)<threshold) { newLeft=oL-newWidth; guideX=oL; snappedX=true; }
            }
            if (!snappedY) {
                if (Math.abs(newTop-oT)<threshold) { newTop=oT; guideY=oT; snappedY=true; }
                else if (Math.abs(cB-oB)<threshold) { newTop=oB-newHeight; guideY=oB; snappedY=true; }
                else if (Math.abs(cCY-oCY)<threshold) { newTop=oCY-newHeight/2; guideY=oCY; snappedY=true; }
                else if (Math.abs(newTop-oB)<threshold) { newTop=oB; guideY=oB; snappedY=true; }
                else if (Math.abs(cB-oT)<threshold) { newTop=oT-newHeight; guideY=oT; snappedY=true; }
            }
        }
        return { newLeft, newTop, snappedX, snappedY, guideX, guideY };
    }

    function updateGuides(el, sx, sy, gx, gy) {
        let gV = document.getElementById('smart-guide-v');
        let gH = document.getElementById('smart-guide-h');
        if (!gV) {
            gV = document.createElement('div'); gV.id='smart-guide-v'; gV.className='smart-guide smart-guide-v';
            gH = document.createElement('div'); gH.id='smart-guide-h'; gH.className='smart-guide smart-guide-h';

        if (gH.parentElement !== el.parentElement) el.parentElement.appendChild(gH);
        gV.style.display = sx ? 'block' : 'none'; if (sx) gV.style.left = gx + 'px';
        gH.style.display = sy ? 'block' : 'none'; if (sy) gH.style.top = gy + 'px';
    }

    // ── DRAG (SINGLE) ──
    function makeDraggable(el) {
        el.onmousedown = (e) => {
            if (e.target.classList.contains('handle')) return;
            if (el.classList.contains('editing')) { e.stopPropagation(); return; }

            // Shift+click: toggle add/remove from multi-selection
            if (e.shiftKey) {
                e.preventDefault(); e.stopPropagation();
                // If there's a single selected element, absorb it into multiSelected first
                if (selectedElement && selectedElement !== el && multiSelected.length === 0) {
                    selectedElement.classList.remove('selected');
                    addToMultiSelected(selectedElement);
                    selectedElement = null;
                    document.getElementById('controls').style.display = 'none';
                }
                // Toggle clicked element
                if (multiSelected.includes(el)) {
                    removeFromMultiSelected(el);
                    if (multiSelected.length === 0) {
                        document.getElementById('no-selection').style.display = 'block';
                        document.getElementById('multi-controls').style.display = 'none';
                    } else {
                        showMultiPanel();
                    }
                } else {
                    addToMultiSelected(el);
                    showMultiPanel();
                }
                return;
            }

            // If currently multi-selected and we click inside the group → drag group
            if (multiSelected.includes(el)) {
                e.preventDefault(); e.stopPropagation();
                const startX = e.clientX, startY = e.clientY;
                const startPositions = multiSelected.map(m => ({ el: m, left: parseFloat(m.style.left)||0, top: parseFloat(m.style.top)||0 }));
                let dragged = false;
                document.onmousemove = (me) => {
                    me.preventDefault(); dragged = true;
                    const dx = (me.clientX - startX) / zoomLevel;
                    const dy = (me.clientY - startY) / zoomLevel;
                    startPositions.forEach(p => {
                        p.el.style.left = (p.left + dx) + 'px';
                        p.el.style.top = (p.top + dy) + 'px';
                    });
                };
                document.onmouseup = () => {
                    document.onmousemove = null; document.onmouseup = null;
                    if (dragged) saveState();
                };
                return;
            }

            e.preventDefault(); e.stopPropagation();
            if (selectedElement !== el) selectElement(el);
            let sx = e.clientX, sy = e.clientY, sl = el.offsetLeft, st = el.offsetTop, dragged = false;
            document.onmouseup = () => { document.onmouseup = null; document.onmousemove = null; hideGuides(); checkPageSwitch(el); if (dragged) saveState(); };
            document.onmousemove = (e) => {
                e.preventDefault(); dragged = true;
                let dx = (e.clientX-sx)/zoomLevel, dy = (e.clientY-sy)/zoomLevel;
                let rawL = Math.round((sl+dx)/SNAP_GRID)*SNAP_GRID;
                let rawT = Math.round((st+dy)/SNAP_GRID)*SNAP_GRID;
                const snap = calculateSnap(el, rawL, rawT, el.offsetWidth, el.offsetHeight);
                el.style.left = snap.newLeft + 'px'; el.style.top = snap.newTop + 'px';
                updateGuides(el, snap.snappedX, snap.snappedY, snap.guideX, snap.guideY);
            };
        };
    }

    // ── MARQUEE (drag-to-select on empty space) ──
    function initMarquee(page) {
        let marqueeEl = null;
        let startX, startY, startPageX, startPageY;
        let active = false;

        page.addEventListener('mousedown', (e) => {
            // Only start marquee if clicking directly on the page (not on an element)
            if (e.target !== page) return;
            if (e.button !== 0) return;
            e.preventDefault();
            active = true;

            const pageRect = page.getBoundingClientRect();
            startX = (e.clientX - pageRect.left) / zoomLevel;
            startY = (e.clientY - pageRect.top) / zoomLevel;
            startPageX = e.clientX;
            startPageY = e.clientY;

            marqueeEl = document.createElement('div');
            marqueeEl.className = 'marquee-select';
            marqueeEl.style.left = startX + 'px';
            marqueeEl.style.top = startY + 'px';
            marqueeEl.style.width = '0px';
            marqueeEl.style.height = '0px';
            marqueeEl.style.display = 'block';
            page.appendChild(marqueeEl);

            // Clear previous selection on fresh drag (not shift)
            if (!e.shiftKey) {
                if (selectedElement) {
                    selectedElement.classList.remove('selected', 'editing');
                    if (selectedElement.classList.contains('text-element')) selectedElement.contentEditable = 'false';
                    selectedElement = null;
                }
                clearMultiSelected();
                document.getElementById('no-selection').style.display = 'block';
                document.getElementById('controls').style.display = 'none';
                document.getElementById('multi-controls').style.display = 'none';
            }

            const onMove = (me) => {
                if (!active) return;
                const pageRect2 = page.getBoundingClientRect();
                const curX = (me.clientX - pageRect2.left) / zoomLevel;
                const curY = (me.clientY - pageRect2.top) / zoomLevel;
                const left = Math.min(startX, curX);
                const top = Math.min(startY, curY);
                const width = Math.abs(curX - startX);
                const height = Math.abs(curY - startY);
                marqueeEl.style.left = left + 'px';
                marqueeEl.style.top = top + 'px';
                marqueeEl.style.width = width + 'px';
                marqueeEl.style.height = height + 'px';
            };

            const onUp = (me) => {
                if (!active) return;
                active = false;
                window.removeEventListener('mousemove', onMove);
                window.removeEventListener('mouseup', onUp);

                // Detect elements within marquee rect
                const pageRect3 = page.getBoundingClientRect();
                const mLeft = parseFloat(marqueeEl.style.left);
                const mTop = parseFloat(marqueeEl.style.top);
                const mRight = mLeft + parseFloat(marqueeEl.style.width);
                const mBottom = mTop + parseFloat(marqueeEl.style.height);

                if (mRight - mLeft > 3 || mBottom - mTop > 3) {
                    // Only select if marquee has actual size
                    const elements = Array.from(page.querySelectorAll('.element'));
                    const hit = elements.filter(el => {
                        const l = parseFloat(el.style.left) || 0;
                        const t = parseFloat(el.style.top) || 0;
                        const r = l + el.offsetWidth;
                        const b = t + el.offsetHeight;
                        // Intersect check
                        return r > mLeft && l < mRight && b > mTop && t < mBottom;
                    });

                    if (hit.length === 1 && !e.shiftKey) {
                        selectElement(hit[0]);
                    } else if (hit.length > 1) {
                        // If single was selected and shift, absorb it
                        if (selectedElement && !multiSelected.includes(selectedElement)) {
                            selectedElement.classList.remove('selected');
                            addToMultiSelected(selectedElement);
                            selectedElement = null;
                        }
                        hit.forEach(el => addToMultiSelected(el));
                        if (multiSelected.length > 0) {
                            document.getElementById('no-selection').style.display = 'none';
                            document.getElementById('controls').style.display = 'none';
                            showMultiPanel();
                        }
                    }
                }

                if (marqueeEl && marqueeEl.parentElement) marqueeEl.remove();
                marqueeEl = null;
            };

            window.addEventListener('mousemove', onMove);
            window.addEventListener('mouseup', onUp);
        });
    }

    function checkPageSwitch(el) {
        document.querySelectorAll('.resume-page').forEach(page => {
            const er = el.getBoundingClientRect(), pr = page.getBoundingClientRect();
            if (er.top > pr.top && er.top < pr.bottom && el.parentElement !== page) {
                page.appendChild(el);
                el.style.top = ((er.top-pr.top)/zoomLevel) + 'px';
            }
        });
    }

    // ── RESIZE ──
    function makeResizable(el) {
        el.querySelectorAll('.handle').forEach(handle => {
            handle.onmousedown = (e) => {
                e.preventDefault(); e.stopPropagation();
                const dir = handle.className.split('handle-')[1];
                const sW = el.offsetWidth, sH = el.offsetHeight, sX = e.clientX, sY = e.clientY, sL = el.offsetLeft, sT = el.offsetTop;
                const onMove = (me) => {
                    const dx = (me.clientX-sX)/zoomLevel, dy = (me.clientY-sY)/zoomLevel;
                    if (dir.includes('e')) { const nW = Math.max(1,sW+dx); el.style.width=nW+'px'; if(selectedElement===el) document.getElementById('elem-width').value=Math.round(nW); }
                    if (dir.includes('s')) { const nH = Math.max(1,sH+dy); el.style.height=nH+'px'; if(selectedElement===el) document.getElementById('elem-height').value=Math.round(nH); }
                    if (dir.includes('w')) { const nW=sW-dx; if(nW>=1){ el.style.width=nW+'px'; el.style.left=(sL+dx)+'px'; if(selectedElement===el) document.getElementById('elem-width').value=Math.round(nW); } }
                    if (dir.includes('n')) { const nH=sH-dy; if(nH>=1){ el.style.height=nH+'px'; el.style.top=(sT+dy)+'px'; if(selectedElement===el) document.getElementById('elem-height').value=Math.round(nH); } }
                };
                const onUp = () => { window.removeEventListener('mousemove', onMove); window.removeEventListener('mouseup', onUp); saveState(); };
                window.addEventListener('mousemove', onMove); window.addEventListener('mouseup', onUp);
            };
        });
    }

    // ── BIND ELEMENT ──
    function bindElementEvents(el) {
        if (el.classList.contains('text-element')) {
            el.contentEditable = "false";
            el.addEventListener('dblclick', (e) => {
                if (isMultiMode()) return; // no editing in multi-mode
                e.stopPropagation();
                el.contentEditable = "true";
                el.classList.add('editing');
                el.focus();
                syncToolbar();
            });
            el.addEventListener('blur', () => saveState());
            el.addEventListener('keyup', () => { if (el.classList.contains('editing')) syncToolbar(); });
            el.addEventListener('mouseup', () => { if (el.classList.contains('editing')) syncToolbar(); });
        }
        makeDraggable(el);
        makeResizable(el);
        el.addEventListener('mousedown', (e) => {
            if (e.target.classList.contains('handle')) return;
            if (el.classList.contains('editing')) { e.stopPropagation(); return; }
            e.stopPropagation();
            // Shift-click handled in makeDraggable, but in case mousedown fires here too:
            if (!e.shiftKey && !multiSelected.includes(el)) {
                if (selectedElement !== el) selectElement(el);
            }
        });
    }

    // ── KEYBOARD SHORTCUTS ──
    document.addEventListener('keydown', (e) => {
        const cmd = e.ctrlKey || e.metaKey;
        const isTyping = selectedElement && selectedElement.classList.contains('editing');

        if (cmd && e.key.toLowerCase() === 'a' && !isTyping) {
            e.preventDefault();
            // Select all elements on active page
            const page = getActivePage();
            const elements = Array.from(page.querySelectorAll('.element'));
            if (selectedElement) { selectedElement.classList.remove('selected'); selectedElement = null; }
            clearMultiSelected();
            elements.forEach(el => addToMultiSelected(el));
            if (multiSelected.length > 0) showMultiPanel();
            return;
        }

        if (cmd && e.key.toLowerCase() === 'z' && !isTyping) { e.preventDefault(); if (e.shiftKey) redo(); else undo(); return; }
        if (cmd && e.key.toLowerCase() === 'y' && !isTyping) { e.preventDefault(); redo(); return; }

        // Copy
        if (cmd && e.key.toLowerCase() === 'c' && !isTyping) {
            e.preventDefault();
            if (isMultiMode()) {
                clipboardMulti = multiSelected.map(el => { const c = el.cloneNode(true); c.classList.remove('multi-selected','selected','editing'); c.contentEditable = "false"; return c; });
                clipboardElement = null;
            } else if (selectedElement) {
                clipboardElement = selectedElement.cloneNode(true);
                clipboardElement.classList.remove('selected','editing');
                clipboardElement.contentEditable = "false";
                clipboardMulti = [];
            }
        }

        // Paste
        if (cmd && e.key.toLowerCase() === 'v' && !isTyping) {
            e.preventDefault();
            if (clipboardMulti.length > 0) {
                // clear current selection
                if (selectedElement) { selectedElement.classList.remove('selected'); selectedElement = null; }
                clearMultiSelected();
                document.getElementById('controls').style.display = 'none';
                clipboardMulti.forEach(src => {
                    const newEl = src.cloneNode(true);
                    newEl.style.left = (parseInt(newEl.style.left)+20)+'px';
                    newEl.style.top = (parseInt(newEl.style.top)+20)+'px';
                    getActivePage().appendChild(newEl);
                    bringToFront(newEl); bindElementEvents(newEl);
                    addToMultiSelected(newEl);
                });
                showMultiPanel();
                // update clipboard so next paste offsets again
                clipboardMulti = multiSelected.map(el => { const c = el.cloneNode(true); c.classList.remove('multi-selected'); return c; });
                saveState();
            } else if (clipboardElement) {
                const newEl = clipboardElement.cloneNode(true);
                newEl.style.left = (parseInt(newEl.style.left)+20)+'px';
                newEl.style.top = (parseInt(newEl.style.top)+20)+'px';
                getActivePage().appendChild(newEl);
                bringToFront(newEl); bindElementEvents(newEl); selectElement(newEl);
                clipboardElement = newEl.cloneNode(true);
                clipboardElement.classList.remove('selected','editing');
                saveState();
            }
        }

        // Delete
        if ((e.key === 'Delete' || e.key === 'Backspace') && !isTyping) {
            e.preventDefault();
            if (isMultiMode()) {
                deleteMultiSelected();
            } else if (selectedElement) {
                deleteSelected();
            }
        }

        // Arrow keys
        if (!isTyping && ['ArrowUp','ArrowDown','ArrowLeft','ArrowRight'].includes(e.key)) {
            if (!selectedElement && multiSelected.length === 0) return;
            e.preventDefault();
            const step = e.shiftKey ? 10 : 1;
            const targets = isMultiMode() ? multiSelected : [selectedElement];
            targets.forEach(el => {
                let l = parseFloat(el.style.left)||0, t = parseFloat(el.style.top)||0;
                if (e.key==='ArrowUp') t-=step;
                if (e.key==='ArrowDown') t+=step;
                if (e.key==='ArrowLeft') l-=step;
                if (e.key==='ArrowRight') l+=step;
                el.style.left = l+'px'; el.style.top = t+'px';
            });
            arrowMoved = true;
        }

        // Escape: collapse multi to deselect
        if (e.key === 'Escape' && !isTyping) {
            deselectAll(null);
        }
    });

    document.addEventListener('keyup', (e) => {
        if (['ArrowUp','ArrowDown','ArrowLeft','ArrowRight'].includes(e.key) && arrowMoved) { saveState(); arrowMoved = false; }
    });

    async function saveToArchives() {
        const userId = localStorage.getItem('userId');
        if (!userId) return alert("Log in to save!");
        if (selectedElement) { selectedElement.classList.remove('selected','editing'); if (selectedElement.classList.contains('text-element')) selectedElement.contentEditable = "false"; }
        clearMultiSelected();
        const docName = document.getElementById('doc-name-input').value.trim() || "Untitled Design";
        const urlParams = new URLSearchParams(window.location.search);
        const existingId = urlParams.get("id");
        const payload = {
            userId, resumeId: existingId,
            personalInfo: { name: docName, email: "custom@proresume.com" },
            templateType: "custom",
            templateSpecificInputs: {
                htmlContent: document.getElementById('page-1').innerHTML,
                isCustom: true,
                canvasHeight: document.getElementById('canvas-height-input').value
            }
        };
        try {
            const res = await fetch("http://localhost:5000/api/resume", { method:"POST", headers:{"Content-Type":"application/json"}, body:JSON.stringify(payload) });
            const result = await res.json();
            if (res.ok) { alert("Saved to archives!"); if (!existingId && result.data?._id) window.history.replaceState(null,"",`/customize.html?id=${result.data._id}`); }
            else alert("Server Error: " + result.error);
        } catch(e) { alert("Error connecting to server."); }
    }

    // ── UTILITY ──
    function rgbToHex(rgb) {
        if (!rgb || rgb.includes('rgba(0,0,0,0)')) return '#000000';
        const res = rgb.match(/\d+/g);
        if (!res) return '#000000';
        return "#" + ((1<<24)+(+res[0]<<16)+(+res[1]<<8)+ +res[2]).toString(16).slice(1);
    }

    // ── LOAD ──
    window.addEventListener('DOMContentLoaded', () => {
        const urlParams = new URLSearchParams(window.location.search);
        const id = urlParams.get("id"), mode = urlParams.get("mode");

        // Init marquee on page
        document.querySelectorAll('.resume-page').forEach(page => initMarquee(page));

        if (id) {
            loadResume(id).then(() => saveState());
        } else if (mode === 'temp') {
            const tempHtml = localStorage.getItem("tempStudioHtml");
            const tempHeight = localStorage.getItem("tempStudioHeight");
            if (tempHtml) {
                document.getElementById('page-1').innerHTML = tempHtml;
                if (tempHeight) { document.getElementById('canvas-height-input').value = tempHeight; updateCanvasHeight(tempHeight); }
                document.getElementById('page-1').querySelectorAll('.element').forEach(el => initElement(el));
                localStorage.removeItem("tempStudioHtml"); localStorage.removeItem("tempStudioHeight");
                saveState();
            }
        } else {
            document.querySelectorAll('.element').forEach(el => initElement(el));
            updateCanvasHeight(1123);
            saveState();
        }
    });

    async function loadResume(id) {
        try {
            const res = await fetch(`http://localhost:5000/api/resume/${id}`);
            const data = await res.json();
            if (data && data.templateSpecificInputs?.htmlContent) {
                const page = document.getElementById('page-1');
                page.innerHTML = data.templateSpecificInputs.htmlContent;
                if (data.personalInfo?.name) document.getElementById('doc-name-input').value = data.personalInfo.name;
                if (data.templateSpecificInputs.canvasHeight) { document.getElementById('canvas-height-input').value = data.templateSpecificInputs.canvasHeight; updateCanvasHeight(data.templateSpecificInputs.canvasHeight); }
                page.querySelectorAll('.element').forEach(el => initElement(el));
                initMarquee(page);
            }
        } catch (err) { console.error("Load error:", err); alert("Failed to load saved data."); }
    }
