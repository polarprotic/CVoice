        /* ---- Mouse Glow ---- */
        const mouseGlow = document.getElementById('mouse-glow');
        window.addEventListener('mousemove', (e) => {
            if(mouseGlow && window.innerWidth > 900) {
                mouseGlow.style.opacity = '1';
                mouseGlow.style.left = e.clientX + 'px';
                mouseGlow.style.top = e.clientY + 'px';
            }
        });
        window.addEventListener('mouseleave', () => {
            if(mouseGlow) mouseGlow.style.opacity = '0';
        });

        /* ---- Notification ---- */
        const notificationBox = document.getElementById('notification-box');
        let notifTimeout;
        function showNotification(msg, type = 'error') {
            const errIcon = `<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><circle cx="12" cy="12" r="10"/><line x1="12" y1="8" x2="12" y2="12"/><line x1="12" y1="16" x2="12.01" y2="16"/></svg>`;
            const okIcon  = `<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M22 11.08V12a10 10 0 1 1-5.93-9.14"/><polyline points="22 4 12 14.01 9 11.01"/></svg>`;
            notificationBox.innerHTML = `${type === 'error' ? errIcon : okIcon}<span>${msg}</span>`;
            notificationBox.className = `notification-box ${type}`;
            clearTimeout(notifTimeout);
            notifTimeout = setTimeout(() => { notificationBox.classList.add('hidden'); notificationBox.classList.remove(type); }, 4000);
        }

        /* ---- Live preview sync ---- */
        function syncPreview(input, origEl, textEl) {
            const has = input.value.trim().length > 0;
            origEl.style.opacity = has ? '0' : '1';
            textEl.style.opacity = has ? '1' : '0';
            textEl.innerText = has ? input.value : '';
        }
        ['name','role','company'].forEach(k => {
            const inp  = document.getElementById('live-' + k);
            const orig = document.getElementById('orig-' + k);
            const txt  = document.getElementById('text-' + k);
            if(inp && orig && txt) {
                inp.addEventListener('input', () => syncPreview(inp, orig, txt));
            }
        });

        /* ---- Toggle logic ---- */
        let isLogin = false;
        const nameGroup            = document.getElementById('name-group');
        const confirmPasswordGroup = document.getElementById('confirm-password-group');
        const formTitle            = document.getElementById('form-title');
        const formSubtitle         = document.getElementById('form-subtitle');
        const submitBtn            = document.getElementById('submit-btn');
        const toggleTextMain       = document.getElementById('toggle-text-main');
        const toggleActionText     = document.getElementById('toggle-action-text');

        document.getElementById('toggle-mode').addEventListener('click', () => {
            isLogin = !isLogin;
            notificationBox.classList.add('hidden');
            document.body.classList.toggle('login-mode', isLogin);

            nameGroup.classList.toggle('hidden-field', isLogin);
            confirmPasswordGroup.classList.toggle('hidden-field', isLogin);
            document.getElementById('name-input').required = !isLogin;

            if (!isLogin) {
                ['live-name','live-role','live-company'].forEach(id => {
                    const el = document.getElementById(id);
                    if(el) { el.value = ''; el.dispatchEvent(new Event('input')); }
                });
            }

            [formTitle, formSubtitle, submitBtn].forEach(el => {
                el.classList.remove('fade-text'); void el.offsetWidth; el.classList.add('fade-text');
            });

            formTitle.innerText        = isLogin ? 'Welcome Back'                           : 'Create Account';
            formSubtitle.innerText     = isLogin ? 'Log in to continue building your legacy.' : 'Start building your professional legacy.';
            submitBtn.innerText        = isLogin ? 'Sign In'                                : 'Continue';
            toggleTextMain.innerText   = isLogin ? "Don't have an account?"                 : 'Already have an account?';
            toggleActionText.innerText = isLogin ? 'Sign up'                                : 'Log in';
        });

        /* ---- Form submit ---- */
        document.getElementById('auth-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            const name            = document.getElementById('name-input').value;
            const email           = document.getElementById('email-input').value;
            const password        = document.getElementById('password-input').value;
            const confirmPassword = document.getElementById('confirm-password-input').value;

            if (!isLogin && password !== confirmPassword) { showNotification('Passwords do not match!', 'error'); return; }

            const endpoint = isLogin ? '/api/auth/login' : '/api/auth/register';
            const payload  = isLogin ? { email, password } : { name, email, password };
            const origText = submitBtn.innerText;
            submitBtn.innerText = 'Processing...'; submitBtn.disabled = true;

            document.getElementById('loader-text').innerText = isLogin ? 'Authenticating...' : 'Creating Account...';
            document.getElementById('global-loader').classList.add('active');

            try {
                const res    = await fetch(endpoint, { method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify(payload) });
                const result = await res.json();
                if (res.ok) {
                    localStorage.setItem('token',     result.token);
                    localStorage.setItem('userId',    result.userId);
                    localStorage.setItem('userName',  result.name);
                    localStorage.setItem('userEmail', email);
                    window.location.href = '/dashboard.html';
                } else {
                    document.getElementById('global-loader').classList.remove('active');
                    showNotification(result.error || 'Authentication failed.', 'error');
                    submitBtn.innerText = origText; submitBtn.disabled = false;
                }
            } catch {
                document.getElementById('global-loader').classList.remove('active');
                showNotification('Could not connect to the server.', 'error');
                submitBtn.innerText = origText; submitBtn.disabled = false;
            }
        });

        /* ---- OAuth ---- */
        function loginWith(provider) {
            if (provider === 'google') {
                const client = google.accounts.oauth2.initTokenClient({
                    client_id: '380873723496-q5avp286nv5ka2g9gs0tpnnuskio65bh.apps.googleusercontent.com',
                    scope: 'https://www.googleapis.com/auth/userinfo.profile https://www.googleapis.com/auth/userinfo.email',
                    callback: async (response) => {
                        if (response && response.access_token) {
                            document.getElementById('loader-text').innerText = 'Syncing Google Profile...';
                            document.getElementById('global-loader').classList.add('active');
                            try {
                                const profileRes = await fetch('https://www.googleapis.com/oauth2/v3/userinfo', { headers:{ Authorization:`Bearer ${response.access_token}` } });
                                const profile = await profileRes.json();
                                localStorage.setItem('token',     response.access_token);
                                localStorage.setItem('userName',  profile.name);
                                localStorage.setItem('userEmail', profile.email);
                                if (profile.picture) localStorage.setItem('userPhoto', profile.picture);
                                localStorage.setItem('userId', BigInt(profile.sub).toString(16).padStart(24,'0'));
                                window.location.href = 'dashboard.html';
                            } catch {
                                document.getElementById('global-loader').classList.remove('active');
                                showNotification('Failed to get Google profile details.', 'error');
                            }
                        }
                    },
                    error_callback: () => showNotification('Google login was cancelled or failed.', 'error')
                });
                client.requestAccessToken();
            } else if (provider === 'github') {
                window.location.href = '/api/auth/github';
            }
        }
