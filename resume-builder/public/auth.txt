    <script>
        /* ---- CONSTELLATION NETWORK GRAPH ---- */
        const canvas = document.getElementById('network-canvas');
        const ctx = canvas.getContext('2d');
        let particlesArray = [];
        
        function setCanvasSize() {
            canvas.width = canvas.offsetWidth;
            canvas.height = canvas.offsetHeight;
        }
        
        let netMouse = { x: null, y: null, radius: 150 };
        const leftPanel = document.getElementById('left-panel');
        
        leftPanel.addEventListener('mousemove', (e) => {
            const rect = canvas.getBoundingClientRect();
            netMouse.x = e.clientX - rect.left;
            netMouse.y = e.clientY - rect.top;
        });
        
        leftPanel.addEventListener('mouseleave', () => {
            netMouse.x = null;
            netMouse.y = null;
        });
        
        window.addEventListener('resize', () => {
            setCanvasSize();
            initParticles();
        });
        
        class Particle {
            constructor(x, y, dx, dy, size) {
                this.x = x; this.y = y;
                this.dx = dx; this.dy = dy;
                this.size = size;
                this.baseX = this.x; this.baseY = this.y;
            }
            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2, false);
                ctx.fillStyle = 'rgba(196, 85, 118, 0.8)';
                ctx.fill();
            }
            update() {
                if (this.x > canvas.width || this.x < 0) this.dx = -this.dx;
                if (this.y > canvas.height || this.y < 0) this.dy = -this.dy;
                
                if (netMouse.x != null) {
                    let dx = netMouse.x - this.x;
                    let dy = netMouse.y - this.y;
                    let distance = Math.sqrt(dx * dx + dy * dy);
                    if (distance < netMouse.radius) {
                        const force = (netMouse.radius - distance) / netMouse.radius;
                        this.x += (dx / distance) * force * 0.5;
                        this.y += (dy / distance) * force * 0.5;
                    } else {
                        this.x += this.dx;
                        this.y += this.dy;
                    }
                } else {
                    this.x += this.dx;
                    this.y += this.dy;
                }
                this.draw();
            }
        }
        
        function initParticles() {
            particlesArray = [];
            setCanvasSize();
            let numNodes = (canvas.width * canvas.height) / 12000; 
            for (let i = 0; i < numNodes; i++) {
                let size = Math.random() * 1.5 + 0.5;
                let x = Math.random() * canvas.width;
                let y = Math.random() * canvas.height;
                let dx = (Math.random() - 0.5) * 0.5;
                let dy = (Math.random() - 0.5) * 0.5;
                particlesArray.push(new Particle(x, y, dx, dy, size));
            }
        }
        
        function animateParticles() {
            requestAnimationFrame(animateParticles);
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            for (let i = 0; i < particlesArray.length; i++) {
                particlesArray[i].update();
            }
            connectParticles();
        }
        
        function connectParticles() {
            let opacityValue = 1;
            for (let a = 0; a < particlesArray.length; a++) {
                for (let b = a; b < particlesArray.length; b++) {
                    let dx = particlesArray[a].x - particlesArray[b].x;
                    let dy = particlesArray[a].y - particlesArray[b].y;
                    let distance = dx * dx + dy * dy;
                    if (distance < 10000) { 
                        opacityValue = 1 - (distance / 10000);
                        ctx.strokeStyle = `rgba(158, 59, 86, ${opacityValue * 0.5})`;
                        ctx.lineWidth = 1;
                        ctx.beginPath();
                        ctx.moveTo(particlesArray[a].x, particlesArray[a].y);
                        ctx.lineTo(particlesArray[b].x, particlesArray[b].y);
                        ctx.stroke();
                    }
                }
            }
        }
        
        setTimeout(() => {
            initParticles();
            animateParticles();
        }, 100);

                client.requestAccessToken();
            } else if (provider === 'github') {
                window.location.href = '/api/auth/github';
            }
