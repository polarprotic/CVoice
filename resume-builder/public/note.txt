    function initAll() {
        const glow = document.getElementById("cursor-glow");
        const dot = document.getElementById("cursor-dot");

        const moveCursor = (e) => {
            if (glow && dot) {
                glow.style.left = e.clientX + "px";
                glow.style.top = e.clientY + "px";
                dot.style.left = e.clientX + "px";
                dot.style.top = e.clientY + "px";
            }
        };
        
        document.removeEventListener("mousemove", moveCursor);
        document.addEventListener("mousemove", moveCursor);

        const storedName = localStorage.getItem('userName');
        const greetingElement = document.getElementById("user-greeting");
        if (storedName && greetingElement) {
            const cleanName = storedName.trim().split(' ')[0];
            const capitalized = cleanName.charAt(0).toUpperCase() + cleanName.slice(1);
            greetingElement.innerText = `Welcome, ${capitalized}`;
        } else if (greetingElement) {
            greetingElement.innerText = `Welcome!`;
        }

        document.onmousedown = () => {
            if(dot) dot.style.transform = "translate(-50%, -50%) scale(1.5)";
        };
        document.onmouseup = () => {
            if(dot) dot.style.transform = "translate(-50%, -50%) scale(1)";
        };
    }

    async function analyzeResume() {
      const job = document.getElementById("dream-job").value;
      const btn = document.getElementById("ai-analyze-btn");
      const resultArea = document.getElementById("ai-result-area");
      
      const scoreNumber = document.getElementById("ai-score-number");
      const scoreRing = document.getElementById("ai-score-ring");
      const atsNumber = document.getElementById("ai-ats-number");
      const atsRing = document.getElementById("ai-ats-ring");
      
      const suggestionsDiv = document.getElementById("ai-suggestions");

      if (!job) return alert("Please enter your dream job!");

      btn.innerText = "Analyzing...";
      btn.disabled = true;

      try {
          const resumeText = document.getElementById("preview").innerText;

          const response = await fetch("http://localhost:5000/api/ai/analyze", {
              method: "POST",
              headers: { "Content-Type": "application/json" },
              body: JSON.stringify({ jobTitle: job, resumeData: resumeText }),
          });

          if (!response.ok) throw new Error("Server responded with an error");

          const data = await response.json();
          console.log("📥 AI Data:", data);

          if (resultArea) resultArea.style.display = "block";

          const circumference = 377; // Math for 140px size

          // --- 1. JOB MATCH SCORE ---
          const rawScore = parseInt(data.score || data.jobScore || 0);
          if (scoreNumber && scoreRing) {
            scoreNumber.innerText = rawScore;
            const hue = (rawScore / 100) * 120;
            const color = `hsl(${hue}, 85%, 50%)`;

            scoreRing.style.stroke = color;
            scoreNumber.style.color = color;
            scoreNumber.style.textShadow = `0 0 10px ${color}`;

            const offset = circumference - (rawScore / 100) * circumference;
            scoreRing.style.transition = "none";
            scoreRing.style.strokeDashoffset = circumference;
            void scoreRing.offsetWidth;

            setTimeout(() => {
              scoreRing.style.transition = "stroke-dashoffset 1.5s cubic-bezier(0.4, 0, 0.2, 1), stroke 1.5s ease";
              scoreRing.style.strokeDashoffset = offset;
            }, 50);
          }

          // --- 2. ATS SCORE ---
          const rawAts = parseInt(data.atsScore || data.ats_score || data.ATS || 0);
          if (atsNumber && atsRing) {
            atsNumber.innerText = rawAts;
            const atsHue = (rawAts / 100) * 120;
            const atsColor = `hsl(${atsHue}, 85%, 50%)`;

            atsRing.style.stroke = atsColor;
            atsNumber.style.color = atsColor;
            atsNumber.style.textShadow = `0 0 10px ${atsColor}`;

            const atsOffset = circumference - (rawAts / 100) * circumference;
            atsRing.style.transition = "none";
            atsRing.style.strokeDashoffset = circumference;
            void atsRing.offsetWidth;

            setTimeout(() => {
              atsRing.style.transition = "stroke-dashoffset 1.5s cubic-bezier(0.4, 0, 0.2, 1), stroke 1.5s ease";
              atsRing.style.strokeDashoffset = atsOffset;
            }, 50);
          }

          // --- 3. SUGGESTIONS ---
          if (suggestionsDiv) {
            if (data.suggestions) {
              suggestionsDiv.innerHTML = data.suggestions
                .replace(/\n/g, "<br>")
                .replace(/\*\*(.*?)\*\*/g, "<strong>$1</strong>");
            } else {
              suggestionsDiv.innerHTML = "No specific suggestions found.";
            }
          }

          if (resultArea) resultArea.scrollIntoView({ behavior: "smooth" });
      } catch (err) {
          console.error("Audit Error:", err);
          alert("Connection to AI server failed.");
      } finally {
          if (btn) {
              btn.innerText = "Audit";
              btn.disabled = false;
          }
      }
    }

      function esc(str) {
        if (!str) return "";
        return String(str)
          .replace(/&/g, "&amp;")
          .replace(/</g, "&lt;")
          .replace(/>/g, "&gt;")
          .replace(/"/g, "&quot;");
      }

      function formatText(text) {
        if (!text) return "";
        return esc(text).replace(/\*\*(.*?)\*\*/g, "<strong>$1</strong>");
      }

      function formatDescToLi(desc) {
        if (!desc) return "";
        return desc.split('\n')
          .map(line => line.trim())
          .filter(line => line.length > 0)
          .map(line => `<li>${formatText(line)}</li>`)
          .join('');
      }

      async function saveResumeToDB() {
        const userId = localStorage.getItem("userId");
        const urlParams = new URLSearchParams(window.location.search);
        const existingResumeId = urlParams.get("id");

        if (!state.name || !state.email) {
          alert("Name and Email are required to save!");
          return;
        }

        const payload = {
          userId: userId,
          resumeId: existingResumeId,
          personalInfo: {
            name: state.name,
            email: state.email,
            phone: state.phone,
          },
          templateType: state.template,
          templateSpecificInputs: state,
        };

        try {
          const response = await fetch("http://localhost:5000/api/resume", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(payload),
          });

          const result = await response.json();

          if (response.ok) {
            alert("✅ Resume saved successfully!");
            if (!existingResumeId && result.data && result.data._id) {
              window.history.replaceState(
                null,
                "",
                `/editor.html?id=${result.data._id}`
              );
            }
          } else {
            alert("❌ Server Error: " + result.error);
          }
        } catch (error) {
          console.error("Save error:", error);
          alert("❌ Network Error: Is your server running on Port 5000?");
        }
      }

      async function loadResumeData(id) {
        try {
          const response = await fetch(`http://localhost:5000/api/resume/${id}`);
          if (!response.ok) throw new Error("Failed to load");

          const data = await response.json();

          if (data && data.templateSpecificInputs) {
            state = data.templateSpecificInputs;
            normalizeState();

            if (state.template) {
              document.getElementById("template-select").value = state.template;
              updateTemplate(state.template);
            }

            renderInputs();
            render();
          }
        } catch (error) {
          console.error("Error loading resume:", error);
          alert("Failed to load your saved resume data.");
        }
      }

      window.addEventListener("DOMContentLoaded", () => {
        const urlParams = new URLSearchParams(window.location.search);
        const requestedTemplate = urlParams.get("template");
        const resumeId = urlParams.get("id");

        if (resumeId) {
          loadResumeData(resumeId);
        } else if (requestedTemplate) {
          document.getElementById("template-select").value = requestedTemplate;
          state.template = requestedTemplate;
          updateTemplate(requestedTemplate);
        } else {
          updateTemplate(document.getElementById("template-select").value);
        }

        renderInputs();
        render();
      });

      async function convertToStudio() {
          if (!localStorage.getItem("userId")) return alert("Please log in to use the Customizer Studio.");

          const preview = document.getElementById("preview");
          const btn = event.currentTarget || document.querySelector('[onclick="convertToStudio()"]');
          const originalText = btn.innerText;
          btn.innerText = "Exporting...";
          btn.disabled = true;

          // 1. Strip the animation BEFORE cloning to prevent scaling bugs
          preview.classList.remove("animating");
          preview.style.animation = "none";
          preview.style.transform = "none"; // Force 100% scale

          // Force browser to recalculate layout at true size
          void preview.offsetWidth;
          
          const parentRect = preview.getBoundingClientRect();
          let htmlStr = "";
          let zIndex = 10;
          let bgZIndex = 1;

          const getStyles = (el) => window.getComputedStyle(el);

          // Add base backgrounds based on template
          if (state.template === 'modern') {
              htmlStr += `<div class="element shape-element" style="left: 0px; top: 0px; width: 278px; height: 100%; background-color: #292929; z-index: 0;"></div>`;
          } else if (state.template === 'academic') {
              htmlStr += `<div class="element shape-element" style="left: 0px; top: 0px; width: 100%; height: 90px; background-color: transparent; z-index: 0;"></div>`;
          } else if (state.template === 'attorney') {
              htmlStr += `<div class="element shape-element" style="left: 0px; bottom: 0px; width: 100%; height: 90px; background-color: transparent; z-index: 0;"></div>`;
          }

          // Measure the LIVE DOM node instead of a hidden wrapper clone to prevent flexbox squishing
          const walkDOM = (node) => {
              if (node.nodeType === 1) {
                  const tag = node.tagName.toLowerCase();
                  if (['script', 'style', 'svg', 'path', 'circle', 'rect', 'line'].includes(tag)) return;
                  
                  const styles = getStyles(node);
                  const r = node.getBoundingClientRect();

                  if (r.width === 0 || r.height === 0 || styles.display === 'none' || styles.visibility === 'hidden') return;

                  let l = r.left - parentRect.left;
                  let t = r.top - parentRect.top;

                  // 1. Extract Images
                  if (tag === 'img' || node.classList.contains('pfp')) {
                      htmlStr += `<div class="element image-element" style="left: ${l}px; top: ${t}px; width: ${r.width}px; height: ${r.height}px; border-radius: ${styles.borderRadius}; background-image: url('${node.src || state.photo}'); background-size: cover; z-index: ${zIndex++}; border: ${styles.borderWidth} solid ${styles.borderColor};"></div>`;
                      return; 
                  }

                  // 2. Extract UI Dots (Modern Template)
                  if (node.classList.contains('timeline-dot') || node.classList.contains('side-dot')) {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t}px; width: ${r.width}px; height: ${r.height}px; border-radius: 50%; background-color: ${styles.backgroundColor}; border: ${styles.borderWidth} solid ${styles.borderColor}; z-index: ${zIndex++};"></div>`;
                      return; 
                  }

                  // 3. Extract Background Shapes (Sidebars, Topbars)
                  const bg = styles.backgroundColor;
                  if (bg !== 'rgba(0, 0, 0, 0)' && bg !== 'transparent' && !node.classList.contains('resume-page')) {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t}px; width: ${r.width}px; height: ${r.height}px; border-radius: ${styles.borderRadius}; background-color: ${bg}; border: ${styles.borderWidth} solid ${styles.borderColor}; z-index: ${bgZIndex++};"></div>`;
                  }

                  // 4. Extract Lines (Borders)
                  if (parseFloat(styles.borderBottomWidth) > 0 && styles.borderBottomStyle !== 'none') {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t + r.height - parseFloat(styles.borderBottomWidth)}px; width: ${r.width}px; height: ${parseFloat(styles.borderBottomWidth)}px; background-color: ${styles.borderBottomColor}; z-index: 5;"></div>`;
                  }
                  if (parseFloat(styles.borderLeftWidth) > 0 && styles.borderLeftStyle !== 'none' && !node.classList.contains('side-timeline') && !node.classList.contains('timeline')) {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t}px; width: ${parseFloat(styles.borderLeftWidth)}px; height: ${r.height}px; background-color: ${styles.borderLeftColor}; z-index: 5;"></div>`;
                  }

                  // 5. Extract Text Blocks Safely
                  let containsBlock = false;
                  let hasText = false;
                  
                  for (let child of node.childNodes) {
                      if (child.nodeType === 3 && child.nodeValue.trim().length > 0) hasText = true;
                      if (child.nodeType === 1) {
                          const d = getStyles(child).display;
                          if (d === 'block' || d === 'flex' || d === 'grid' || d === 'table' || d === 'list-item') {
                              containsBlock = true;
                              break;
                          }
                      }
                  }

                  if (!containsBlock && (hasText || node.innerText.trim() !== '')) {
                      let content = node.innerHTML;
                      if (tag === 'li') content = "• " + content; 
                      
                      // Explicitly capture all typography so it survives the transfer exactly as designed
                      const fontFam = styles.fontFamily.replace(/"/g, "'");
                      const ls = styles.letterSpacing === 'normal' ? '0px' : styles.letterSpacing;
                      const tt = styles.textTransform;
                      const fs = styles.fontStyle;
                      const fw = styles.fontWeight;
                      const fSize = styles.fontSize;
                      const lh = styles.lineHeight;
                      const ta = styles.textAlign;
                      const color = styles.color;
                      
                      // We give the width an extra 15px buffer to stop text from wrapping prematurely 
                      htmlStr += `<div class="element text-element" contenteditable="false" spellcheck="false" style="left: ${l}px; top: ${t}px; width: ${r.width + 15}px; font-size: ${fSize}; font-weight: ${fw}; font-style: ${fs}; color: ${color}; font-family: ${fontFam}; text-align: ${ta}; line-height: ${lh}; letter-spacing: ${ls}; text-transform: ${tt}; white-space: pre-wrap; z-index: ${zIndex++};">${content}</div>`;
                      return; // Stop recursion for this branch
                  }

                  // Recurse children
                  for (let child of node.childNodes) {
                      walkDOM(child);
                  }
              }
          };

          walkDOM(preview);

          // Restore the preview scaling back to fit the window
          preview.style.transform = "";
          preview.style.transition = "";

          // Export to Customize via LocalStorage
          try {
              localStorage.setItem("tempStudioHtml", htmlStr);
              localStorage.setItem("tempStudioHeight", "1123"); 
              window.location.href = "customize.html?mode=temp";
          } catch (e) {
              console.error(e);
              alert("Network error while exporting.");
              btn.innerText = originalText;
              btn.disabled = false;
          }
      }

    <script>
        lucide.createIcons();
        let selectedElement = null;
        let clipboardElement = null;
        let pageCount = 1;
        let imageUploadTarget = 'global';
        let highestZIndex = 10;
        
        // 🚨 GLOBAL ZOOM AND GRID VARS 🚨
        const SNAP_GRID = 10;
        let zoomLevel = 1;
        
        // 🚨 UNDO/REDO STATE SYSTEM 🚨
        let historyStack = [];
        let historyIndex = -1;

        // 🚨 ZOOM ENGINE LOGIC 🚨
        function updateZoom(val) {
            zoomLevel = parseFloat(val);
            document.getElementById('zoom-val').innerText = Math.round(zoomLevel * 100) + '%';
            
            document.querySelectorAll('.resume-page').forEach(page => {
                page.style.transform = `scale(${zoomLevel})`;
                
                // Ensures container scrollbar height matches visually scaled element
                const heightDiff = (page.offsetHeight * zoomLevel) - page.offsetHeight;
                page.style.marginBottom = heightDiff > 0 ? `${heightDiff + 50}px` : '0px';
            });
        }

        function changeZoom(delta) {
            let val = parseFloat(document.getElementById('zoom-slider').value) + delta;
            val = Math.max(0.5, Math.min(1.5, val)); // limit zoom bounds
            document.getElementById('zoom-slider').value = val;
            updateZoom(val);
        }

        function saveState() {
            if (historyIndex < historyStack.length - 1) {
                historyStack = historyStack.slice(0, historyIndex + 1);
            }
            
            const clone = document.getElementById('page-1').cloneNode(true);
            clone.querySelectorAll('.selected, .editing').forEach(el => {
                el.classList.remove('selected', 'editing');
                if (el.classList.contains('text-element')) el.contentEditable = "false";
            });
            // remove guides from history state
            clone.querySelectorAll('.smart-guide').forEach(el => el.remove());
            
            historyStack.push(clone.innerHTML);
            
            if (historyStack.length > 30) {
                historyStack.shift();
            } else {
                historyIndex++;
            }
        }

        function undo() {
            if (historyIndex > 0) {
                deselect({target: document.getElementById('workspace')});
                historyIndex--;
                restoreState(historyStack[historyIndex]);
            }
        }

        function redo() {
            if (historyIndex < historyStack.length - 1) {
                deselect({target: document.getElementById('workspace')});
                historyIndex++;
                restoreState(historyStack[historyIndex]);
            }
        }

        function restoreState(htmlString) {
            const page = document.getElementById('page-1');
            page.innerHTML = htmlString;
            const loadedElements = page.querySelectorAll('.element');
            loadedElements.forEach(el => initElement(el));
        }
        
        let currentRange = null;
        document.addEventListener('selectionchange', () => {
            const sel = window.getSelection();
            if (sel.rangeCount > 0 && selectedElement && selectedElement.contains(sel.anchorNode)) {
                currentRange = sel.getRangeAt(0);
            }
        });

        document.getElementById('inspector').addEventListener('mousedown', (e) => {
            if (e.target.tagName !== 'INPUT' && e.target.tagName !== 'SELECT' && e.target.tagName !== 'TEXTAREA') {
                e.preventDefault(); 
            }
        });

        function updateCanvasHeight(val) {
            const pages = document.querySelectorAll('.resume-page');
            pages.forEach(page => { page.style.height = val + 'px'; });
            const sidebarBg = document.querySelector('.shape-element[style*="left: 0px"]');
            if(sidebarBg) sidebarBg.style.height = '100%';
            updateZoom(document.getElementById('zoom-slider').value); // Re-calculate margin scaling
        }

        function resetCanvasHeight() {
            document.getElementById('canvas-height-input').value = 1123;
            updateCanvasHeight(1123);
        }

        function addNewPage() {
            pageCount++;
            const newPage = document.createElement('div');

            document.getElementById('color-controls').style.display = 'block';

        function applyTextFormat(command, val = null, save = true) {
            if (!selectedElement || !selectedElement.classList.contains('text-element')) return;

            const isEditing = selectedElement.classList.contains('editing');
            const hasSelection = currentRange && currentRange.toString().length > 0 && selectedElement.contains(currentRange.commonAncestorContainer);

            if (isEditing) {
                selectedElement.focus();
                if (hasSelection) {
                    const sel = window.getSelection();
                    sel.removeAllRanges();
                    sel.addRange(currentRange);
                }

                if (command === 'fontSizePx') {
                    document.execCommand('fontSize', false, '7');
                    selectedElement.querySelectorAll('font[size="7"]').forEach(font => {
                        font.removeAttribute('size');
                        font.style.fontSize = val + 'px';
                    });
                } else if (command === 'fontFamily') {
                    document.execCommand('fontName', false, val.split(',')[0].replace(/['"]/g, ''));
                } else if (command === 'foreColor') {
                    document.execCommand('foreColor', false, val);
                } else {
                    document.execCommand(command, false, val);
                }
                
                const newSel = window.getSelection();
                if (newSel.rangeCount > 0) currentRange = newSel.getRangeAt(0);

            } else {
                if (command === 'fontSizePx') updateStyle('fontSize', val + 'px', false);
                if (command === 'fontFamily') updateStyle('fontFamily', val, false);
                if (command === 'foreColor') updateStyle('color', val, false);
                if (command === 'bold') {
                    const w = window.getComputedStyle(selectedElement).fontWeight;
                    updateStyle('fontWeight', w >= 600 ? '400' : '700', false);
                }
                if (command === 'italic') {
                    const i = window.getComputedStyle(selectedElement).fontStyle;
                    updateStyle('fontStyle', i === 'italic' ? 'normal' : 'italic', false);
                }
                if (command === 'underline') {
                    const u = window.getComputedStyle(selectedElement).textDecorationLine;
                    updateStyle('textDecoration', u === 'underline' ? 'none' : 'underline', false);
                }
                if (command === 'insertUnorderedList' || command === 'insertOrderedList') {
                    selectedElement.contentEditable = "true";
                    selectedElement.focus();
                    document.execCommand('selectAll', false, null);
                    document.execCommand(command, false, val);
                    selectedElement.contentEditable = "false";
                    window.getSelection().removeAllRanges();
                }
            }
            if (save) saveState();
        }

        // 🚨 SMART GUIDES LOGIC 🚨
        function calculateSnap(el, newLeft, newTop, newWidth, newHeight) {
            const threshold = 5; // Snap pixel radius
            let snappedX = false; let snappedY = false;
            let guideX = -1; let guideY = -1;

            // Check all other elements on the same page
            const rects = Array.from(el.parentElement.querySelectorAll('.element')).filter(e => e !== el && !e.classList.contains('handle'));
            
            for (let other of rects) {
                const oLeft = parseFloat(other.style.left) || 0;
                const oTop = parseFloat(other.style.top) || 0;
                const oWidth = other.offsetWidth;
                const oHeight = other.offsetHeight;
                const oRight = oLeft + oWidth;
                const oBottom = oTop + oHeight;
                const oCenterX = oLeft + oWidth / 2;
                const oCenterY = oTop + oHeight / 2;

                const cLeft = newLeft;
                const cRight = newLeft + newWidth;
                const cCenterX = newLeft + newWidth / 2;
                const cTop = newTop;
                const cBottom = newTop + newHeight;
                const cCenterY = newTop + newHeight / 2;

                // X-Axis Snapping
                if (!snappedX) {
                    if (Math.abs(cLeft - oLeft) < threshold) { newLeft = oLeft; guideX = oLeft; snappedX = true; }
                    else if (Math.abs(cRight - oRight) < threshold) { newLeft = oRight - newWidth; guideX = oRight; snappedX = true; }
                    else if (Math.abs(cCenterX - oCenterX) < threshold) { newLeft = oCenterX - newWidth / 2; guideX = oCenterX; snappedX = true; }
                    else if (Math.abs(cLeft - oRight) < threshold) { newLeft = oRight; guideX = oRight; snappedX = true; }
                    else if (Math.abs(cRight - oLeft) < threshold) { newLeft = oLeft - newWidth; guideX = oLeft; snappedX = true; }
                }

                // Y-Axis Snapping
                if (!snappedY) {
                    if (Math.abs(cTop - oTop) < threshold) { newTop = oTop; guideY = oTop; snappedY = true; }
                    else if (Math.abs(cBottom - oBottom) < threshold) { newTop = oBottom - newHeight; guideY = oBottom; snappedY = true; }
                    else if (Math.abs(cCenterY - oCenterY) < threshold) { newTop = oCenterY - newHeight / 2; guideY = oCenterY; snappedY = true; }
                    else if (Math.abs(cTop - oBottom) < threshold) { newTop = oBottom; guideY = oBottom; snappedY = true; }
                    else if (Math.abs(cBottom - oTop) < threshold) { newTop = oTop - newHeight; guideY = oTop; snappedY = true; }
                }
            }

            return { newLeft, newTop, snappedX, snappedY, guideX, guideY };
        }

        function updateGuides(el, sx, sy, gx, gy) {

        // 🚨 ZOOM SCALED DRAGGING ENGINE + SMART GUIDES 🚨
        function makeDraggable(el) {
            el.onmousedown = (e) => {
                if (e.target.classList.contains('handle')) return;
                if (el.classList.contains('editing')) {
                    e.stopPropagation(); 
                    return; 
                }

                e.preventDefault(); 
                e.stopPropagation(); 
                
                if (selectedElement !== el) {
                    selectElement(el);
                }

                let startX = e.clientX;
                let startY = e.clientY;
                let startLeft = el.offsetLeft;
                let startTop = el.offsetTop;
                let dragged = false;

                document.onmouseup = () => { 
                    document.onmouseup = null; 
                    document.onmousemove = null; 
                    hideGuides(); 
                    checkPageSwitch(el); 
                    if(dragged) saveState();
                };
                
                document.onmousemove = (e) => {
                    e.preventDefault(); 
                    dragged = true;
                    
                    // DIVIDE MOVEMENT BY ZOOM LEVEL TO KEEP MOUSE SYNCHRONIZED
                    let dx = (e.clientX - startX) / zoomLevel;
                    let dy = (e.clientY - startY) / zoomLevel;
                    
                    let rawLeft = Math.round((startLeft + dx) / SNAP_GRID) * SNAP_GRID;
                    let rawTop = Math.round((startTop + dy) / SNAP_GRID) * SNAP_GRID;
                    
                    const w = el.offsetWidth;
                    const h = el.offsetHeight;

                    const snap = calculateSnap(el, rawLeft, rawTop, w, h);
                    
                    el.style.left = snap.newLeft + "px";
                    el.style.top = snap.newTop + "px";

                    updateGuides(el, snap.snappedX, snap.snappedY, snap.guideX, snap.guideY);
                };
            };
        }

        function checkPageSwitch(el) {
            const pages = document.querySelectorAll('.resume-page');
            const elRect = el.getBoundingClientRect();
            pages.forEach(page => {
                const pageRect = page.getBoundingClientRect();
                if (elRect.top > pageRect.top && elRect.top < pageRect.bottom) {
                    if (el.parentElement !== page) {
                        page.appendChild(el);
                        // Adjust Y placement to fix zoom offset scaling bugs
                        el.style.top = ((elRect.top - pageRect.top) / zoomLevel) + "px";
                    }
                }
            });
        }

        // 🚨 ZOOM SCALED RESIZING ENGINE 🚨
        function makeResizable(el) {
            const handles = el.querySelectorAll('.handle');
            handles.forEach(handle => {
                handle.onmousedown = (e) => {
                    e.preventDefault(); e.stopPropagation();
                    const dir = handle.className.split('handle-')[1];
                    
                    const startW = el.offsetWidth; const startH = el.offsetHeight;
                    const startX = e.clientX; const startY = e.clientY;
                    const startLeft = el.offsetLeft; const startTop = el.offsetTop;

                    const onMove = (moveEvent) => {
                        // DIVIDE MOVEMENT BY ZOOM LEVEL
                        const dx = (moveEvent.clientX - startX) / zoomLevel;
                        const dy = (moveEvent.clientY - startY) / zoomLevel;

                        if (dir.includes('e')) {
                            const newW = Math.max(1, startW + dx);
                            el.style.width = newW + 'px'; 
                            if(selectedElement === el) document.getElementById('elem-width').value = Math.round(newW);
                        }
                        if (dir.includes('s')) {
                            const newH = Math.max(1, startH + dy);
                            el.style.height = newH + 'px'; 
                            if(selectedElement === el) document.getElementById('elem-height').value = Math.round(newH);
                        }
                        
                        if (dir.includes('w')) {
                            const newW = startW - dx;
                            if (newW >= 1) { 
                                el.style.width = newW + 'px'; 
                                el.style.left = startLeft + dx + 'px'; 
                                if(selectedElement === el) document.getElementById('elem-width').value = Math.round(newW);
                            }
                        }
                        if (dir.includes('n')) {
                            const newH = startH - dy;
                            if (newH >= 1) { 
                                el.style.height = newH + 'px'; 
                                el.style.top = startTop + dy + 'px'; 
                                if(selectedElement === el) document.getElementById('elem-height').value = Math.round(newH);
                            }
                        }
                    };

                    const onUp = () => { 
                        window.removeEventListener('mousemove', onMove); 
                        window.removeEventListener('mouseup', onUp); 
                        saveState();
                    };
                    window.addEventListener('mousemove', onMove); 
                    window.addEventListener('mouseup', onUp);
                };
            });
        }

        let arrowMoved = false;
        document.addEventListener('keydown', (e) => {
            const cmdKey = e.ctrlKey || e.metaKey;
            const isTyping = selectedElement && selectedElement.classList.contains('editing');
            
            if (cmdKey && e.key.toLowerCase() === 'z' && !isTyping) {
                e.preventDefault();
                if (e.shiftKey) redo(); else undo();
                return;
            }
            if (cmdKey && e.key.toLowerCase() === 'y' && !isTyping) {
                e.preventDefault();
                redo();
                return;
            }

            if (cmdKey && e.key.toLowerCase() === 'a' && !isTyping) e.preventDefault(); 
            if (cmdKey && e.key.toLowerCase() === 'c' && !isTyping) { 
                if (selectedElement) { 
                    e.preventDefault(); 
                    clipboardElement = selectedElement.cloneNode(true); 
                    clipboardElement.classList.remove('selected', 'editing'); 
                    clipboardElement.contentEditable = "false";
                } 
            }
            if (cmdKey && e.key.toLowerCase() === 'v' && !isTyping) { 
                if (clipboardElement) { 
                    e.preventDefault(); 
                    const newEl = clipboardElement.cloneNode(true); 
                    
                    newEl.style.left = (parseInt(newEl.style.left) + 20) + 'px'; 
                    newEl.style.top = (parseInt(newEl.style.top) + 20) + 'px'; 
                    
                    const targetPage = document.querySelector('.resume-page:hover') || document.getElementById('page-1'); 
                    targetPage.appendChild(newEl); 
                    bringToFront(newEl); bindElementEvents(newEl); selectElement(newEl); 
                    clipboardElement = newEl.cloneNode(true); 
                    clipboardElement.classList.remove('selected', 'editing'); 
                    saveState();
                } 
            }
            if ((e.key === 'Delete' || e.key === 'Backspace') && selectedElement && !isTyping) { 
                e.preventDefault(); deleteSelected(); 
            }

            if (selectedElement && !isTyping && ['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].includes(e.key)) {
                e.preventDefault();
                let left = parseFloat(selectedElement.style.left) || 0;
                let top = parseFloat(selectedElement.style.top) || 0;
                const step = e.shiftKey ? 10 : 1;
                
                if (e.key === 'ArrowUp') top -= step;
                if (e.key === 'ArrowDown') top += step;
                if (e.key === 'ArrowLeft') left -= step;
                if (e.key === 'ArrowRight') left += step;
                
                selectedElement.style.left = left + 'px';
                selectedElement.style.top = top + 'px';
                arrowMoved = true;
            }
        });

        document.addEventListener('keyup', (e) => {
            if (['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].includes(e.key) && arrowMoved) {
                saveState();
                arrowMoved = false;
            }
        });

        // 🚨 PREVENTS PDF BLUR BY RESETTING ZOOM 🚨
        function downloadPDF() {
            if (selectedElement) {
                selectedElement.classList.remove('selected', 'editing');
                if (selectedElement.classList.contains('text-element')) selectedElement.contentEditable = "false";
            }
            
            const element = document.getElementById('page-1');
            const originalTransform = element.style.transform;
            
            // Snap back to exactly 100% for perfect 4K render quality
            element.style.transform = 'scale(1)';
            
            const opt = { 
                margin: 0, 
                filename: 'custom-resume.pdf', 
                image: { type: 'jpeg', quality: 1 }, 
                html2canvas: { 
                    scale: 2, 
                    useCORS: true,
                    backgroundColor: '#ffffff' 
                }, 
                jsPDF: { 
                    unit: 'px', 
                    format: [794, parseInt(element.style.height || 1123)], 
                    orientation: 'portrait' 
                } 
            };
            
            const originalShadow = element.style.boxShadow;
            element.style.boxShadow = 'none';
            
            setTimeout(() => {
                html2pdf().set(opt).from(element).save().then(() => {
                    element.style.boxShadow = originalShadow;
                    // Reset user's selected zoom
                    element.style.transform = originalTransform;
                });
            }, 150); // Pause briefly to let DOM paint the un-zoomed format
        }

        async function saveToArchives() {
            const userId = localStorage.getItem('userId');
            if(!userId) return alert("Log in to save!");

            if (selectedElement) {
                selectedElement.classList.remove('selected', 'editing');
                if (selectedElement.classList.contains('text-element')) selectedElement.contentEditable = "false";
            }
            
            const docName = document.getElementById('doc-name-input').value.trim() || "Untitled Design";
            const urlParams = new URLSearchParams(window.location.search);
            const existingId = urlParams.get("id");

            const payload = {
                userId: userId, 
                resumeId: existingId,
                personalInfo: { 
                    name: docName, 
                    email: "custom@solaristools.com" 
                },
                templateType: "custom", 
                templateSpecificInputs: { 
                    htmlContent: document.getElementById('page-1').innerHTML, 
                    isCustom: true,
                    canvasHeight: document.getElementById('canvas-height-input').value
                }
            };

            try {
                const response = await fetch("http://localhost:5000/api/resume", { 
                    method: "POST", 
                    headers: { "Content-Type": "application/json" }, 
                    body: JSON.stringify(payload) 
                });
                
                const result = await response.json();

                if (response.ok) {
                    alert("Masterpiece saved to archives!");
                    if (!existingId && result.data?._id) {
                        window.history.replaceState(null, "", `/customize.html?id=${result.data._id}`);
                    }
                } else {
                    alert("Server Error: " + result.error);
                }
            } catch(e) {

        function rgbToHex(rgb) {
            if(!rgb || rgb.includes('rgba(0,0,0,0)')) return '#000000';
            const res = rgb.match(/\d+/g);
            if(!res) return '#000000';
            return "#" + ((1 << 24) + (+res[0] << 16) + (+res[1] << 8) + +res[2]).toString(16).slice(1);
        }

        function initElement(el) {
            if(el.classList.contains('text-element')) {
                el.contentEditable = "false";
                el.classList.remove('editing');
            }
            addResizeHandles(el);
            bindElementEvents(el);
            
            const z = parseInt(window.getComputedStyle(el).zIndex);
            if (!isNaN(z) && z > highestZIndex) highestZIndex = z;
        }

       window.addEventListener('DOMContentLoaded', () => {
            const urlParams = new URLSearchParams(window.location.search);
            const id = urlParams.get("id");
            const mode = urlParams.get("mode"); 
            
            if (id) {
                loadResume(id).then(() => saveState());
            } else if (mode === 'temp') {
                const tempHtml = localStorage.getItem("tempStudioHtml");
                const tempHeight = localStorage.getItem("tempStudioHeight");
                
                if (tempHtml) {
                    document.getElementById('page-1').innerHTML = tempHtml;
                    
                    if (tempHeight) {
                        document.getElementById('canvas-height-input').value = tempHeight;
                        updateCanvasHeight(tempHeight);
                    }

                    const loadedElements = document.getElementById('page-1').querySelectorAll('.element');
                    loadedElements.forEach(el => initElement(el));
                    
                    localStorage.removeItem("tempStudioHtml");
                    localStorage.removeItem("tempStudioHeight");
                    
                    saveState();
                }
            } else {
                document.querySelectorAll('.element').forEach(el => initElement(el));
                updateCanvasHeight(1123);
                const sidebarBg = document.querySelector('.shape-element[style*="left: 0px"]');
                if(sidebarBg) sidebarBg.style.height = '100%';
                saveState();
            }
        });

        async function loadResume(id) {
            try {
                const response = await fetch(`http://localhost:5000/api/resume/${id}`);
                const data = await response.json();

                if (data && data.templateSpecificInputs?.htmlContent) {
                    const page = document.getElementById('page-1');
                    
                    page.innerHTML = data.templateSpecificInputs.htmlContent;
                    
                    if (data.personalInfo && data.personalInfo.name) {
                        document.getElementById('doc-name-input').value = data.personalInfo.name;
                    }
                    
                    if(data.templateSpecificInputs.canvasHeight) {
                        document.getElementById('canvas-height-input').value = data.templateSpecificInputs.canvasHeight;
                        updateCanvasHeight(data.templateSpecificInputs.canvasHeight);
                    }

                    const loadedElements = page.querySelectorAll('.element');
                    loadedElements.forEach(el => initElement(el));
                }
            } catch (error) {
                console.error("Load error:", error);
                alert("Failed to load saved data.");
            }
        }

    async function analyzeResume() {
      const job = document.getElementById("dream-job").value;
      const btn = document.getElementById("ai-analyze-btn");
      const resultArea = document.getElementById("ai-result-area");
      
      const scoreNumber = document.getElementById("ai-score-number");
      const scoreRing = document.getElementById("ai-score-ring");
      const atsNumber = document.getElementById("ai-ats-number");
      const atsRing = document.getElementById("ai-ats-ring");
      
      const suggestionsDiv = document.getElementById("ai-suggestions");

      if (!job) return alert("Please enter your dream job!");

      btn.innerText = "Analyzing...";
      btn.disabled = true;

      try {
          const resumeText = document.getElementById("preview").innerText;

          const response = await fetch("http://localhost:5000/api/ai/analyze", {
              method: "POST",
              headers: { "Content-Type": "application/json" },
              body: JSON.stringify({ jobTitle: job, resumeData: resumeText }),
          });

          if (!response.ok) throw new Error("Server responded with an error");

          const data = await response.json();
          console.log("📥 AI Data:", data);

          if (resultArea) resultArea.style.display = "block";

          const circumference = 377; // Math for 140px size

          // --- 1. JOB MATCH SCORE ---
          const rawScore = parseInt(data.score || data.jobScore || 0);
          if (scoreNumber && scoreRing) {
            scoreNumber.innerText = rawScore;
            const hue = (rawScore / 100) * 120;
            const color = `hsl(${hue}, 85%, 50%)`;

            scoreRing.style.stroke = color;
            scoreNumber.style.color = color;
            scoreNumber.style.textShadow = `0 0 10px ${color}`;

            const offset = circumference - (rawScore / 100) * circumference;
            scoreRing.style.transition = "none";
            scoreRing.style.strokeDashoffset = circumference;
            void scoreRing.offsetWidth;

            setTimeout(() => {
              scoreRing.style.transition = "stroke-dashoffset 1.5s cubic-bezier(0.4, 0, 0.2, 1), stroke 1.5s ease";
              scoreRing.style.strokeDashoffset = offset;
            }, 50);
          }

          // --- 2. ATS SCORE ---
          const rawAts = parseInt(data.atsScore || data.ats_score || data.ATS || 0);
          if (atsNumber && atsRing) {
            atsNumber.innerText = rawAts;
            const atsHue = (rawAts / 100) * 120;
            const atsColor = `hsl(${atsHue}, 85%, 50%)`;

            atsRing.style.stroke = atsColor;
            atsNumber.style.color = atsColor;
            atsNumber.style.textShadow = `0 0 10px ${atsColor}`;

            const atsOffset = circumference - (rawAts / 100) * circumference;
            atsRing.style.transition = "none";
            atsRing.style.strokeDashoffset = circumference;
            void atsRing.offsetWidth;

            setTimeout(() => {
              atsRing.style.transition = "stroke-dashoffset 1.5s cubic-bezier(0.4, 0, 0.2, 1), stroke 1.5s ease";
              atsRing.style.strokeDashoffset = atsOffset;
            }, 50);
          }

          // --- 3. SUGGESTIONS ---
          if (suggestionsDiv) {
            if (data.suggestions) {
              suggestionsDiv.innerHTML = data.suggestions
                .replace(/\n/g, "<br>")
                .replace(/\*\*(.*?)\*\*/g, "<strong>$1</strong>");
            } else {
              suggestionsDiv.innerHTML = "No specific suggestions found.";
            }
          }

          if (resultArea) resultArea.scrollIntoView({ behavior: "smooth" });
      } catch (err) {
          console.error("Audit Error:", err);
          alert("Connection to AI server failed.");
      } finally {
          if (btn) {
              btn.innerText = "Audit";
              btn.disabled = false;
          }
      }
    }

      function esc(str) {
        if (!str) return "";
        return String(str)
          .replace(/&/g, "&amp;")
          .replace(/</g, "&lt;")
          .replace(/>/g, "&gt;")
          .replace(/"/g, "&quot;");
      }

      function formatText(text) {
        if (!text) return "";
        return esc(text).replace(/\*\*(.*?)\*\*/g, "<strong>$1</strong>");
      }

      function formatDescToLi(desc) {
        if (!desc) return "";
        return desc.split('\n')
          .map(line => line.trim())
          .filter(line => line.length > 0)
          .map(line => {
              // 🚨 FIX: Remove leading bullets or dashes if user types them, prevents double bulleting 🚨
              let cleanedLine = line.replace(/^[•\-]\s*/, '');
              return `<li>${formatText(cleanedLine)}</li>`;
          })
          .join('');
      }

      async function saveResumeToDB() {
        const userId = localStorage.getItem("userId");
        const urlParams = new URLSearchParams(window.location.search);
        const existingResumeId = urlParams.get("id");

        if (!state.name || !state.email) {
          alert("Name and Email are required to save!");
          return;
        }

        const payload = {
          userId: userId,
          resumeId: existingResumeId,
          personalInfo: {
            name: state.name,
            email: state.email,
            phone: state.phone,
          },
          templateType: state.template,
          templateSpecificInputs: state,
        };

        try {
          const response = await fetch("http://localhost:5000/api/resume", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(payload),
          });

          const result = await response.json();

          if (response.ok) {
            alert("✅ Resume saved successfully!");
            if (!existingResumeId && result.data && result.data._id) {
              window.history.replaceState(
                null,
                "",
                `/editor.html?id=${result.data._id}`
              );
            }
          } else {
            alert("❌ Server Error: " + result.error);
          }
        } catch (error) {
          console.error("Save error:", error);
          alert("❌ Network Error: Is your server running on Port 5000?");
        }
      }

      async function loadResumeData(id) {
        try {
          const response = await fetch(`http://localhost:5000/api/resume/${id}`);
          if (!response.ok) throw new Error("Failed to load");

          const data = await response.json();

          if (data && data.templateSpecificInputs) {
            state = data.templateSpecificInputs;
            normalizeState();

            if (state.template) {
              document.getElementById("template-select").value = state.template;
              updateTemplate(state.template);
            }

            renderInputs();
            render();
          }
        } catch (error) {
          console.error("Error loading resume:", error);
          alert("Failed to load your saved resume data.");
        }
      }

      window.addEventListener("DOMContentLoaded", () => {
        const urlParams = new URLSearchParams(window.location.search);
        const requestedTemplate = urlParams.get("template");
        const resumeId = urlParams.get("id");

        if (resumeId) {
          loadResumeData(resumeId);
        } else if (requestedTemplate) {
          document.getElementById("template-select").value = requestedTemplate;
          state.template = requestedTemplate;
          updateTemplate(requestedTemplate);
        } else {
          updateTemplate(document.getElementById("template-select").value);
        }

        renderInputs();
        render();
      });

      // 🚨 PERFECTED ALGORITHM: 1:1 LIVE DOM FLATTEN 🚨
      async function convertToStudio() {
          if (!localStorage.getItem("userId")) return alert("Please log in to use the Customizer Studio.");

          const preview = document.getElementById("preview");
          const btn = event.currentTarget || document.querySelector('[onclick="convertToStudio()"]');
          const originalText = btn.innerText;
          btn.innerText = "Exporting...";
          btn.disabled = true;

          // 1. Strip the animation BEFORE cloning to prevent scaling bugs
          preview.classList.remove("animating");
          preview.style.animation = "none";
          preview.style.transform = "none"; // Force 100% scale

          // Force browser to recalculate layout at true size
          void preview.offsetWidth;
          
          const parentRect = preview.getBoundingClientRect();
          let htmlStr = "";
          let zIndex = 10;
          let bgZIndex = 1;

          const getStyles = (el) => window.getComputedStyle(el);

          // Add base backgrounds based on template
          if (state.template === 'modern') {
              htmlStr += `<div class="element shape-element" style="left: 0px; top: 0px; width: 278px; height: 100%; background-color: #292929; z-index: 0;"></div>`;
          } else if (state.template === 'academic') {
              htmlStr += `<div class="element shape-element" style="left: 0px; top: 0px; width: 100%; height: 90px; background-color: transparent; z-index: 0;"></div>`;
          } else if (state.template === 'attorney') {
              htmlStr += `<div class="element shape-element" style="left: 0px; bottom: 0px; width: 100%; height: 90px; background-color: transparent; z-index: 0;"></div>`;
          }

          // Measure the LIVE DOM node instead of a hidden wrapper clone to prevent flexbox squishing
          const walkDOM = (node) => {
              if (node.nodeType === 1) {
                  const tag = node.tagName.toLowerCase();
                  if (['script', 'style', 'svg', 'path', 'circle', 'rect', 'line'].includes(tag)) return;
                  
                  const styles = getStyles(node);
                  const r = node.getBoundingClientRect();

                  if (r.width === 0 || r.height === 0 || styles.display === 'none' || styles.visibility === 'hidden') return;

                  let l = r.left - parentRect.left;
                  let t = r.top - parentRect.top;

                  // 1. Extract Images
                  if (tag === 'img' || node.classList.contains('pfp')) {
                      htmlStr += `<div class="element image-element" style="left: ${l}px; top: ${t}px; width: ${r.width}px; height: ${r.height}px; border-radius: ${styles.borderRadius}; background-image: url('${node.src || state.photo}'); background-size: cover; z-index: ${zIndex++}; border: ${styles.borderWidth} solid ${styles.borderColor};"></div>`;
                      return; 
                  }

                  // 2. Extract UI Dots (Modern Template)
                  if (node.classList.contains('timeline-dot') || node.classList.contains('side-dot')) {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t}px; width: ${r.width}px; height: ${r.height}px; border-radius: 50%; background-color: ${styles.backgroundColor}; border: ${styles.borderWidth} solid ${styles.borderColor}; z-index: ${zIndex++};"></div>`;
                      return; 
                  }

                  // 3. Extract Background Shapes (Sidebars, Topbars)
                  const bg = styles.backgroundColor;
                  if (bg !== 'rgba(0, 0, 0, 0)' && bg !== 'transparent' && !node.classList.contains('resume-page')) {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t}px; width: ${r.width}px; height: ${r.height}px; border-radius: ${styles.borderRadius}; background-color: ${bg}; border: ${styles.borderWidth} solid ${styles.borderColor}; z-index: ${bgZIndex++};"></div>`;
                  }

                  // 4. Extract Lines (Borders)
                  if (parseFloat(styles.borderBottomWidth) > 0 && styles.borderBottomStyle !== 'none') {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t + r.height - parseFloat(styles.borderBottomWidth)}px; width: ${r.width}px; height: ${parseFloat(styles.borderBottomWidth)}px; background-color: ${styles.borderBottomColor}; z-index: 5;"></div>`;
                  }
                  if (parseFloat(styles.borderLeftWidth) > 0 && styles.borderLeftStyle !== 'none' && !node.classList.contains('side-timeline') && !node.classList.contains('timeline')) {
                      htmlStr += `<div class="element shape-element" style="left: ${l}px; top: ${t}px; width: ${parseFloat(styles.borderLeftWidth)}px; height: ${r.height}px; background-color: ${styles.borderLeftColor}; z-index: 5;"></div>`;
                  }

                  // 5. Extract Text Blocks Safely
                  let containsBlock = false;
                  let hasText = false;
                  
                  for (let child of node.childNodes) {
                      if (child.nodeType === 3 && child.nodeValue.trim().length > 0) hasText = true;
                      if (child.nodeType === 1) {
                          const d = getStyles(child).display;
                          if (d === 'block' || d === 'flex' || d === 'grid' || d === 'table' || d === 'list-item') {
                              containsBlock = true;
                              break;
                          }
                      }
                  }

                  if (!containsBlock && (hasText || node.innerText.trim() !== '')) {
                      let content = node.innerHTML;
                      if (tag === 'li') {
                          if (state.template === 'attorney') content = "– " + content;
                          else content = "• " + content;
                      } 
                      
                      // Explicitly capture all typography so it survives the transfer exactly as designed
                      const fontFam = styles.fontFamily.replace(/"/g, "'");
                      const ls = styles.letterSpacing === 'normal' ? '0px' : styles.letterSpacing;
                      const tt = styles.textTransform;
                      const fs = styles.fontStyle;
                      const fw = styles.fontWeight;
                      const fSize = styles.fontSize;
                      const lh = styles.lineHeight;
                      const ta = styles.textAlign;
                      const color = styles.color;
                      
                      // We give the width an extra 15px buffer to stop text from wrapping prematurely 
                      htmlStr += `<div class="element text-element" contenteditable="false" spellcheck="false" style="left: ${l}px; top: ${t}px; width: ${r.width + 15}px; font-size: ${fSize}; font-weight: ${fw}; font-style: ${fs}; color: ${color}; font-family: ${fontFam}; text-align: ${ta}; line-height: ${lh}; letter-spacing: ${ls}; text-transform: ${tt}; white-space: pre-wrap; z-index: ${zIndex++};">${content}</div>`;
                      return; // Stop recursion for this branch
                  }

                  // Recurse children
                  for (let child of node.childNodes) {
                      walkDOM(child);
                  }
              }
          };

          walkDOM(preview);

          // Restore the preview scaling back to fit the window
          preview.style.transform = "";
          preview.style.transition = "";

          // Export to Customize via LocalStorage
          try {
              localStorage.setItem("tempStudioHtml", htmlStr);
              localStorage.setItem("tempStudioHeight", "1123"); 
              window.location.href = "customize.html?mode=temp";
          } catch (e) {
              console.error(e);
              alert("Network error while exporting.");
              btn.innerText = originalText;
              btn.disabled = false;
          }
      }
    </script>
